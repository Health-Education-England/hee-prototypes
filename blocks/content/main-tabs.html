    title: "Tab component",

      {
        label: "Tab one",
        panel: {
        html: tabOneContent
      }
      },
      {
        label: "Tab two",
        panel: {
        html: tabTwoContent
      }
      },
      {
        label: "Tab three",
        panel: {
        html: tabThreeContent

      }
